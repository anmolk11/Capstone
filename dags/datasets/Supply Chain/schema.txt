    "Shipping carriers" VARCHAR(255),
    "Shipping costs" FLOAT,

    Location VARCHAR(255),
    "Lead time" INT,
    "Production volumes" INT,
    "Manufacturing lead time" INT,
    "Manufacturing costs" FLOAT,
    "Inspection results" VARCHAR(255),
    "Defect rates" FLOAT,
    "Transportation modes" VARCHAR(255),
    "Routes" VARCHAR(255),
